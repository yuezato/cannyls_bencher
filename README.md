# cannyls_bencher

# ワークロード記述
```
# 乱数のシード
# シードは非負整数で与えるが、省略可能（省略した場合は Seed: 0;に等しい）
Seed: 42; 

# 100iterationで
# 上から順番に実行。この場合は
# 20件データを書き込み（すべて43バイト）
# 10件データを読み込み
# 30件データを削除
# 39件データを削除（ただし生き残っているデータのうち10%-20%区間から）
# 1件DeleteRangeを発行（99%-100%区間をすべて消そうとする）
Ordered[100] {
  <20%> New(43);
  <10%> Get;
  <30%> Delete;
  <39%> Delete(10, 20);
  <1%> DeleteRange(99, 100);
}

# 200iterationで
# 順不同実行
Unordered[200] {
  <10%> Get;
  <20%> New(43);
  <30%> Delete;
  <39%> Delete(10, 20);
  <1%> DeleteRange(99, 100);
}
```
